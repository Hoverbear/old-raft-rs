/// Since ::std::num::Zero isn't stable, we have our own version.
pub trait Zero {
    fn zero() -> Self;
}

macro_rules! zero_impl(
    ($t:ident, $e:expr) => (
        impl Zero for $t {
            #[inline]
            fn zero() -> $t { $e }
        }
    )
);

zero_impl!(u8,  0);
zero_impl!(u16, 0);
zero_impl!(u32, 0);
zero_impl!(u64, 0);
zero_impl!(i8,  0);
zero_impl!(i16, 0);
zero_impl!(i32, 0);
zero_impl!(i64, 0);
zero_impl!(f32, 0.0);
zero_impl!(f64, 0.0);
